
## Forward Declaration 

// --------------------------------------------
// --------------------------------------------

- **Example**:  Forward Declaration 

#define _CRT_SECURE_NO_WARNINGS
#include <iostream>

class A; // Forward Declaration for class A
class B; // Forward Declaration for class B

class B {
public:
	int x;

	void getdata(int n)	{
		x = n;
	}

	friend int sum(A, B);
};

class A {
public:
	int y;

	void getdata(int m)	{
		y = m;
	}

	friend int sum(A, B);
};

int sum(A m, B n){
	return m.y + n.x;
}

int main()
{
	B b;
	A a;
	
	a.getdata(5);
	b.getdata(4);
	std::cout << "The sum is : " << sum(a, b);
	// The sum is : 9
	
	return 0;
}
